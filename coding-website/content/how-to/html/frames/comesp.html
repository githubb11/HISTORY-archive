add javascript:

Example

<script>
var countDownDate = new Date("Jan 1, 2022 00:00:00">

var x =setInterval(function() {
var now = new Date().getTime();
var disance = countDownDate - now;
var days = Math.floor(disance / (1000 * 60 * 60 * 24));
var hours = Math.floor((disance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
var minutes = Math.floor((disance % (1000 * 60 * 60)) / (1000 * 60));
var seconds = Math.floor((disance % (1000 * 60)) / 1000);
document.getElementById("launch").innerHTML = days + "d " + hours + "h " + minutes +"m " + seconds + "s ";

if (disance < 0) {
clearInterval(x);
document.getElementById("launch").innerHTML = "EXPIRED;"
}
}, 1000 );
</script>
